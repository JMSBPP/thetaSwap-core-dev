\section{Initialization}\label{sec:initialization}

\subsection{Input Parameters}

The CFMM is initialized from:
\begin{itemize}
  \item $B_0 \in [0, 1)$: the initial fee dispersion index (typically $B_0 = 0$
        at the first \texttt{afterAddLiquidity} event);
  \item $L_0 > 0$: the initial liquidity.
\end{itemize}

\subsection{Derived Initial State}

\begin{proposition}[Initial Price]\label{prop:init-price}
The initial price is the odds-ratio mapping:
\begin{equation}\label{eq:init-price}
  p_0 = \frac{B_0}{1 - B_0}
\end{equation}
\end{proposition}

\begin{proposition}[Initial Reserves]\label{prop:init-reserves}
The initial reserves are:
\begin{align}
  x_0 &= L_0 \cdot (1 - B_0) \label{eq:init-x}\\
  y_0 &= L_0 \cdot \bigl(-\ln(1 - B_0) - B_0\bigr) \label{eq:init-y}
\end{align}
\end{proposition}

\begin{proof}
From \eqref{eq:risky-reserve} with liquidity scaling:
\[
  x_0 = L_0 \cdot \frac{1}{1 + p_0}
      = L_0 \cdot \frac{1}{1 + B_0/(1-B_0)}
      = L_0 \cdot \frac{1}{1/(1-B_0)}
      = L_0 \cdot (1 - B_0)
\]

From \eqref{eq:numeraire-reserve} with liquidity scaling:
\begin{align*}
  y_0 &= L_0 \cdot \Bigl(\ln(1 + p_0) - \frac{p_0}{1 + p_0}\Bigr) \\
      &= L_0 \cdot \Bigl(\ln\bigl(\tfrac{1}{1-B_0}\bigr) - \frac{B_0/(1-B_0)}{1/(1-B_0)}\Bigr) \\
      &= L_0 \cdot \bigl(-\ln(1 - B_0) - B_0\bigr) \qedhere
\end{align*}
\end{proof}

\begin{proposition}[Invariant Constant]\label{prop:init-invariant}
The initial invariant evaluates to:
\begin{equation}\label{eq:init-invariant}
  \psi\bigl(\tfrac{x_0}{L_0},\, \tfrac{y_0}{L_0}\bigr) = 0
\end{equation}
\end{proposition}

\begin{proof}
Let $u_0 = x_0/L_0 = 1 - B_0$. Then $y_0/L_0 = -\ln(1-B_0) - B_0 = -\ln(u_0) - (1 - u_0) = u_0 - \ln(u_0) - 1$.
By \eqref{eq:trading-function-explicit}, $y/L = u - \ln(u) - 1$ is exactly the invariant curve. \qedhere
\end{proof}

\subsection{Special Case: $B_0 = 0$}

At the first \texttt{afterAddLiquidity} event, no positions have been removed yet,
so $A_0 = 0$ and $B_0 = 1 - A_0 = 1$. However, $B_0 = 1$ is outside the domain
($p \to \infty$). In practice:

\begin{remark}[Initialization at Genesis]\label{rem:genesis}
When $B_0 = 1$ (no concentration), $p_0 \to \infty$ and $x_0 \to 0$.
The CFMM starts with all reserves in numeraire.
To avoid degenerate initialization, we set $B_0 = 1 - \epsilon$ for a small
$\epsilon > 0$ chosen at deployment, giving:
\begin{align}
  p_0 &= (1 - \epsilon) / \epsilon \\
  x_0 &= L_0 \cdot \epsilon \\
  y_0 &= L_0 \cdot (-\ln(\epsilon) - (1 - \epsilon))
\end{align}
\end{remark}

\subsection{Numerical Verification}

At $B_0 = 0.5$, $L_0 = 100$:
\begin{align*}
  p_0 &= 0.5 / 0.5 = 1.0 \\
  x_0 &= 100 \cdot 0.5 = 50 \\
  y_0 &= 100 \cdot (-\ln(0.5) - 0.5) = 100 \cdot (0.6931 - 0.5) = 19.31
\end{align*}

Verification of LP value:
\[
  p_0 \cdot x_0 + y_0 = 1.0 \cdot 50 + 19.31 = 69.31 = L_0 \cdot \ln(1 + 1.0) = 100 \cdot 0.6931 \;\checkmark
\]

Verification of invariant:
\[
  \psi(50/100,\, 19.31/100) = 0.1931 + \ln(0.5) + 1 - 0.5 = 0.1931 - 0.6931 + 0.5 = 0.0 \;\checkmark
\]
